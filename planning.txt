drop table memoryMeta;
create table memoryMeta(
    	uuid int PRIMARY KEY NOT NULL AUTO_INCREMENT ,
    	chapter int,
    	figure varchar(20),
    	description varchar(150),
    	value0Label varchar(20),
    	value1Label varchar(20),
    	value2Label varchar(20),
    	value3Label varchar(20));
drop table memoryEntries;
create table memoryEntries(
		uuid int PRIMARY KEY NOT NULL AUTO_INCREMENT,
    	figure varchar(20),
    	value0 varchar(100),
    	value1 varchar(100),
    	value2 varchar(100),
    	value3 varchar(100));
